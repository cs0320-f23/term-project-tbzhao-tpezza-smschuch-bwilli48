4   import java.util.ArrayList;

6   import java.util.List;

9   /**
10   * Parse class is used for developers to parse their CSV into a list of objects of their choice It
11   * takes in a Reader with their CSV file path, and a custom row parser that implements
12   * CreatorFromRow and returns an object of their choice
13   *
14   * <p>TO USE: Must create a custom CreatorFromRow class, and pass that in with a reader with CSV
15   * file to a parse object, then run parse. This could be done in a custom main method.
16   *
17   * @param <T> the object type the developer wants their return list to contain
18   */
19  public class Parse<T> {
20    Reader reader;
21    CreatorFromRow<T> creator;
22  
23    /**
24     * Constructor for parse
25     *
26     * @param reader Reader with CSV file path
27     * @param creator custom row parser implementing CreatorFromRow
28     */
29    public Parse(Reader reader, CreatorFromRow<T> rowType) {
30      this.reader = reader;
31      this.creator = rowType;
32    }
33  
34    /**
35     * Main parse method
36     *
37     * @return List<T> where T is developers desired object for parsed csv to contain
38     */
39    public List<T> parse() {
40      BufferedReader bf = new BufferedReader(reader);
41      List<T> rowList = new ArrayList<>();
42      try {
43        String fl = bf.readLine();
44        while (fl != null) {
45          rowList.add(this.creator.create(parseRow(fl)));
46          fl = bf.readLine();
47        }
48      } catch (IOException i) {
49        System.err.println("While parsing, the file path was not found.");
50      } catch (FactoryFailureException e) {
51        System.err.println(
52            "Error parsing the row from a list of string to "
53                + "the desired object with the inputted creator from row.");
54        // for testing purposes
55        return Collections.emptyList();
56      }
57      return rowList;
58    }
59  
60    /**
61     * Method that takes in each line from the CSV, and returns a List<String>, where each String was
62     * seperated by the provided regex Allows row to be passed into the custom CreatorFromRow
63     *
64     * @param row line from CSV as a String
65     * @return List<String> of row split by regex
66     */
67    public List<String> parseRow(String row) {
68      final Pattern regexSplitCSVRow =
69          Pattern.compile(",(?=([^\\\"]*\\\"[^\\\"]*\\\")*(?![^\\\"]*\\\"))");
70  
71      List<String> list = List.of(regexSplitCSVRow.split(row));
72      return list;
73    }
74  }

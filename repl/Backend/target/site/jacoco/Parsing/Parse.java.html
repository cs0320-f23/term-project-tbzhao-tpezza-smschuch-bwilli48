package Parsing;

import java.io.*;
import java.util.ArrayList;
import java.util.List;
import java.util.regex.Pattern;

public class Parse<T> {

  // read line
  // pass to create
  // add created to list
  Reader reader;
  CreatorFromRow<T> rowType;

  public Parse(Reader reader, CreatorFromRow<T> rowType) {
    this.reader = reader;
    this.rowType = rowType;
  }

  public List<T> parse() {
    BufferedReader bf = new BufferedReader(reader);
    List<T> rowList = new ArrayList<>();
    try {
      String fl = bf.readLine();
      while (fl != null) {
        rowList.add(this.rowType.create(parseRow(fl)));
        fl = bf.readLine();
      }
    } catch (IOException i) {
      System.out.println("While parsing, the file path was not found.");
    } catch (FactoryFailureException e) {
      System.out.println(
          "Error parsing the row from a list of string to "
              + "the desired object with the inputted creator from row.");
    }
    return rowList;
  }

  public List<String> parseRow(String row) {
    final Pattern regexSplitCSVRow =
        Pattern.compile(",(?=([^\\\"]*\\\"[^\\\"]*\\\")*(?![^\\\"]*\\\"))");

    List<String> list = List.of(regexSplitCSVRow.split(row));
    return list;
  }
}
